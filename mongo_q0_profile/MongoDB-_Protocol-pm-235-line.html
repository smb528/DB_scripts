337# CursorID's can come in 3 forms:
338#
339# 1. MongoDB::CursorID object (a blessed reference to an 8-byte string)
340# 2. A perl scalar (an integer)
341# 3. A Math::BigInt object (64 bit integer on 32-bit perl)
342#
343# The _pack_cursor_id function converts any of them to a packed Int64 for
344# use in OP_GET_MORE or OP_KILL_CURSORS
345sub _pack_cursor_id {
346 my $cursor_id = shift;
347 if ( ref($cursor_id) eq "MongoDB::_CursorID" ) {
348 $cursor_id = $$cursor_id;
349 }
350 elsif ( ref($cursor_id) eq "Math::BigInt" ) {
351 my $as_hex = $cursor_id->as_hex; # big-endian hex
352 substr( $as_hex, 0, 2, '' ); # remove "0x"
353 my $len = length($as_hex);
354 substr( $as_hex, 0, 0, "0" x ( 16 - $len ) ) if $len < 16; # pad to quad length
355 $cursor_id = pack( "H*", $as_hex ); # packed big-endian
356 $cursor_id = reverse($cursor_id); # reverse to little-endian
357 }
358 elsif (HAS_INT64) {
359 # pack doesn't have endianness modifiers before perl 5.10.
360 # We die during configuration on big-endian platforms on 5.8
361 $cursor_id = pack( $] lt '5.010' ? "q" : "q<", $cursor_id );
362 }
363 else {
364 # we on 32-bit perl *and* have a cursor ID that fits in 32 bits,
365 # so pack it as long and pad out to a quad
366 $cursor_id = pack( $] lt '5.010' ? "l" : "l<", $cursor_id ) . ( "\0" x 4 );
367 }
368
369 return $cursor_id;
370}
