271sub remove {
272 my ( $self, $criteria, $options ) = @_;
273 $options ||= {};
274
275 my $chunks =
276 exists $options->{safe}
277 ? $self->chunks->clone( write_concern => $self->_dynamic_write_concern($options) )
278 : $self->chunks;
279
280 my $files =
281 exists $options->{safe}
282 ? $self->files->clone( write_concern => $self->_dynamic_write_concern($options) )
283 : $self->files;
284
285 if ( $options->{just_one} ) {
286 my $meta = $files->find_one($criteria);
287 $chunks->delete_many( { "files_id" => $meta->{'_id'} } );
288 $files->delete_one( { "_id" => $meta->{'_id'} } );
289 }
290 else {
291 my $cursor = $files->find($criteria);
292 while ( my $meta = $cursor->next ) {
293 $chunks->delete_many( { "files_id" => $meta->{'_id'} } );
294 }
295 $files->delete_many($criteria);
296 }
297 return;

336sub insert {
337 my ($self, $fh, $metadata, $options) = @_;
338 $options ||= {};
339 require DateTime; # lazy load so we don't have overhead if not needed
340
341 MongoDB::UsageError->throw("not a file handle") unless $fh;
342 $metadata = {} unless $metadata && ref $metadata eq 'HASH';
343
344 my $chunks =
345 exists $options->{safe}
346 ? $self->chunks->clone( write_concern => $self->_dynamic_write_concern($options) )
347 : $self->chunks;
348
349 my $files =
350 exists $options->{safe}
351 ? $self->files->clone( write_concern => $self->_dynamic_write_concern($options) )
352 : $self->files;
353
354 my $start_pos = $fh->getpos();
355
356 my $id;
357 if (exists $metadata->{"_id"}) {
358 $id = $metadata->{"_id"};
359 }
360 else {
361 $id = MongoDB::OID->new;
362 }
363
364 my $n = 0;
365 my $length = 0;
366 while ((my $len = $fh->read(my $data, $MongoDB::GridFS::chunk_size)) != 0) {
367 $chunks->insert_one({"files_id" => $id,
368 "n" => $n,
369 "data" => \$data});
370 $n++;
371 $length += $len;
372 }
373
374 $fh->setpos($start_pos);
375
376 my %copy = %{$metadata};
377 # compare the md5 hashes
378 if ($files->write_concern->is_acknowledged) {
379 # get an md5 hash for the file. set the retry flag to 'true' incase the
380 # database, collection, or indexes are missing. That way we can recreate them
381 # retry the md5 calc.
382 my $result = $self->_database->run_command([filemd5 => $id, root => $self->prefix]);
383 $copy{"md5"} = $result->{"md5"};
384
385 my $md5 = Digest::MD5->new;
386 $md5->addfile($fh);
387 $fh->setpos($start_pos);
388 my $digest = $md5->hexdigest;
389 if ($digest ne $result->{md5}) {
390 # cleanup and die
391 $chunks->delete_many({files_id => $id});
392 MongoDB::GridFSError->throw(
393 "md5 hashes don't match: database got $result->{md5}, fs got $digest" );
394 }
395 }
396
397 $copy{"_id"} = $id;
398 $copy{"chunkSize"} = $MongoDB::GridFS::chunk_size;
399 $copy{"uploadDate"} = DateTime->now;
400 $copy{"length"} = $length;
401 return $files->insert_one(\%copy)->inserted_id;
402}

446sub _dynamic_write_concern {
